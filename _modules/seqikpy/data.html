def get_pts2align(path: str):
    """ Deletes the keys from the PTS2ALIGN dictionary."""
    pts_temp = PTS2ALIGN.copy()
    if "_RF" in path:
        del pts_temp["RF_leg"]
    elif "_LF" in path:
        del pts_temp["LF_leg"]
    elif "_RLF" in path or "_LRF" in path:
        del pts_temp["LF_leg"]
        del pts_temp["RF_leg"]

    return pts_temp
